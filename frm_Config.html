unit frm_Variables;

{$mode ObjFPC}{$H+}

interface

uses
  Classes, SysUtils, Forms, Controls, Graphics, Dialogs, ComCtrls, ButtonPanel,
  Menus, StdCtrls, MaskEdit, DateTimePicker, MaskUtils, RegExpr, ark_uDataModule;

type

  { TFormVariables }

  TFormVariables = class(TForm)
    ButtonPanel1: TButtonPanel;
    cmb_StatusVar: TComboBox;
    dtp_FCreacionVar: TDateTimePicker;
    edt_CodigoVar: TEdit;
    edt_DescripcionVar: TEdit;
    edt_ValorVar: TEdit;
    lab_CodigoVar: TLabel;
    lab_DescripcionVar: TLabel;
    lab_Fecha: TLabel;
    lab_ValorVar: TLabel;
    lab_StatusVar: TLabel;
    MainMenuVar: TMainMenu;
    MenuItem1: TMenuItem;
    mit_Borrar: TMenuItem;
    mit_Buscar: TMenuItem;
    mit_Cancelar: TMenuItem;
    mit_Editar: TMenuItem;
    mit_Incluir: TMenuItem;
    tbn_Borras: TToolButton;
    tbn_Buscar: TToolButton;
    tbn_Cancelar: TToolButton;
    tbn_Edit: TToolButton;
    tbn_Incluir: TToolButton;
    tbr_Variables: TToolBar;
    procedure FormCreate(Sender: TObject);
    procedure FormShow(Sender: TObject);
    procedure mit_CancelarClick(Sender: TObject);
    procedure mit_IncluirClick(Sender: TObject);
    procedure tbn_CancelarClick(Sender: TObject);
    procedure tbn_IncluirClick(Sender: TObject);
    procedure CancelButtonClick(Sender: TObject);
  private
    procedure ActivarEdicion;             // Declaración del  método Incluir.
    procedure CancelarEdicion;            // Declaración del  método Cancelar.
  public


  end;

var
  Pfrm_Variables: TFormVariables;

implementation

{$R *.lfm}

{ TFormVariables }
procedure TFormVariables.FormCreate(Sender: TObject);
begin
    // Desactivar todos los TEdit
    edt_CodigoVar.Enabled       := False;
    cmb_StatusVar.Enabled       := False;
    edt_DescripcionVar.Enabled  := False;
    dtp_FCreacionVar.Enabled    := False;
    edt_ValorVar.Enabled        := False;
END;

procedure TFormVariables.ActivarEdicion;
begin
  // Activar todos los TEdit
  edt_CodigoVar.Enabled         := True;
  cmb_StatusVar.Enabled         := True;
  edt_DescripcionVar.Enabled    := True;
  dtp_FCreacionVar.Enabled      := True;
  edt_ValorVar.Enabled          := True;
end;

procedure TFormVariables.CancelarEdicion;
begin
  if MessageDlg('¿Está seguro que desea cancelar?', mtConfirmation, [mbYes, mbNo], 0) = mrYes then
  begin
    // Limpiar todos los TEdit
    edt_CodigoVar.Text          := '';
    cmb_StatusVar.Text          := '';
    edt_DescripcionVar.Text     := '';
    edt_ValorVar.Text           := '';

    // Desactivar todos los TEdit
    edt_CodigoVar.Enabled       := False;
    cmb_StatusVar.Enabled       := False;
    edt_DescripcionVar.Enabled  := False;
    dtp_FCreacionVar.Enabled    := False;
    edt_ValorVar.Enabled        := False;
  end;
end;

procedure TFormVariables.mit_IncluirClick(Sender: TObject);
begin
  ActivarEdicion;
end;

procedure TFormVariables.tbn_IncluirClick(Sender: TObject);
begin
  ActivarEdicion;
end;

procedure TFormVariables.mit_CancelarClick(Sender: TObject);
begin
  CancelarEdicion;
end;

procedure TFormVariables.tbn_CancelarClick(Sender: TObject);
begin
  CancelarEdicion;
end;

procedure TFormVariables.CancelButtonClick(Sender: TObject);
begin
  CancelarEdicion;
end;

procedure TFormVariables.FormShow(Sender: TObject);
begin
     // Asignar la fecha actual del sistema al dtp_FCreacionCat
     dtp_FCreacionVar.Date := Now;

end;


end.
